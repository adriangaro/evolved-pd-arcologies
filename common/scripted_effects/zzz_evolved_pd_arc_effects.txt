tec_pd_arc_generic_city_planet_logic_start = {
	optimize_memory
	tec_ecu_deposits_handler_before = yes
	clear_deposits = yes

	change_pc = $change_pc$

	tec_pd_arc_clean_evolved_modifiers = yes
	tec_pd_arc_clean_environment_engineering = yes
	tec_pd_arc_planet_entity_and_atomsphere_generic = yes

	[[relic]
		tec_pd_arc_clear_evolved_flags = yes
		tec_pd_arc_set_evolved_relic_flag = yes
	]
}

tec_pd_arc_generic_city_planet_logic_finish = {
	optimize_memory
	[[relic]
		add_deposit = d_former_relic_world
		while = {
			count = 4
			solar_system = {
				random_system_ambient_object = {
					limit = { has_ambient_object_flag = relic_world_debris }
					destroy_ambient_object = this
				}
			}
		}
	]
	tec_ecu_ecological_resolution = yes
	tec_ecu_deposits_handler_after = yes
}

tec_pd_arc_terraforming_effect = {
	optimize_memory
	owner = {
		change_variable = {
			which = tec_terraformed_worlds
			value = 1
		}
	}

	if = {
		limit = {
			exists = owner
			owner = {
				has_technology = tech_terrestrial_sculpting
				has_ascension_perk = ap_world_shaper
				OR = {
					has_valid_civic = civic_tec_eco_keepers
					has_valid_civic = civic_agrarian_idyll
				}
			}
			planet = {
				NOR = {
					tec_is_pd_perfect_planets = yes
					tec_is_pd_exotic_planets = yes
				}
			}
		}
		change_pc = pc_gaia
	}
}

tec_pd_arc_demote_industrial_jobs = {
	optimize_memory
	if = {
		limit = { 
			exists = owner
			owner = {
				is_regular_empire = yes
			}
		}
		
		every_owned_pop = {
			limit = {
				planet = {
					is_same_value = root
				}
				OR = {
					has_job = foundry
					has_job = catalytic_technician
					has_job = artisan
					has_job = artificer

					has_job = alloy_drone
					has_job = fabricator
					has_job = catalytic_technician
					has_job = artisan_drone
				}
			}
			add_modifier = {
				modifier = tec_demote_me_senpai
				days = 45
			}
		}
	}
}
